039 /**
040  * Mutable, ordered, fixed sized sequence.
041  *
042  * <p>
043  * <b>Implementation note:</b>
044  * <i>This implementation is not thread safe. All {@link ISeq} and {@link MSeq}
045  * instances created by {@link MSeq#toISeq} and {@link MSeq#subSeq(int)},
046  * respectively, must be protected by the same lock, when they are accessed
047  * (get/set) by different threads.</i>
048  *
049  * @see ISeq
050  *
051  * @author <a href="mailto:franz.wilhelmstoetter@gmx.at">Franz Wilhelmstötter</a>
052  * @since 1.0
053  * @version 3.3
054  */
055 public interface MSeq<T> extends Seq<T>, Copyable<MSeq<T>> {
056 
057     /**
058      * Set the {@code value} at the given {@code index}.
059      *
060      * @param index the index of the new value.
061      * @param value the new value.
062      * @throws IndexOutOfBoundsException if the index is out of range
063      *         {@code (index < 0 || index >= size())}.
064      */
065     public void set(final int index, final T value);
066 
067     /**
068      * Fills the sequence with values of the given iterator.
069      *
070      * @param it the iterator of the values to fill this sequence.
071      * @return {@code this} sequence.
072      */
073     public default MSeq<T> setAll(final Iterator<? extends T> it) {
074         for (int i = 0, n = length(); i < n && it.hasNext(); ++i) {
075             set(i, it.next());
076         }
077         return this;
078     }
079 
080     /**
081      * Fills the sequence with values of the given iterable.
082      *
083      * @param values the values to fill this sequence.
084      * @return {@code this} sequence.
085      */
086     public default MSeq<T> setAll(final Iterable<? extends T> values) {
087         setAll(values.iterator());
088         return this;
089     }
090 
091     /**
092      * Fill the sequence with the given values.
093      *
094      * @param values the first initial values of this sequence
095      * @return {@code this} sequence.
096      */
097     public default MSeq<T> setAll(final T[] values) {
098         for (int i = 0, n = length(); i < n && i < values.length; ++i) {
099             set(i, values[i]);
100         }
101         return this;
102     }
103 
104     /**
105      * Fill the sequence with values generated by the given factory.
106      *
107      * @param supplier the value factory.
108      * @return {@code this} sequence.
109      * @throws NullPointerException if the given {@code factory} is {@code null}.
110      */
111     public default MSeq<T> fill(final Supplier<? extends T> supplier) {
112         for (int i = 0, n = length(); i < n; ++i) {
113             set(i, supplier.get());
114         }
115         return this;
116     }

118     /**
119      * Swap the elements at the two positions.
120      *
121      * @param i the index of the first element.
122      * @param j the index of the second element.
123      * @throws IndexOutOfBoundsException if {@code i < 0 || j >= length()}.
124      */
125     public default void swap(final int i, final int j) {
126         final T temp = get(i);
127         set(i, get(j));
128         set(j, temp);
129     }

131     /**
132      * Swap a given range with a range of the same size with another array.
133      *
134      * <pre>
135      *            start                end
136      *              |                   |
137      * this:  +---+---+---+---+---+---+---+---+---+---+---+---+
138      *              +---------------+
139      *                          +---------------+
140      * other: +---+---+---+---+---+---+---+---+---+---+---+---+
141      *                          |
142      *                      otherStart
143      * </pre>
144      *
145      * @param start the start index of {@code this} range, inclusively.
146      * @param end the end index of {@code this} range, exclusively.
147      * @param other the other array to swap the elements with.
148      * @param otherStart the start index of the {@code other} array.
149      * @throws IndexOutOfBoundsException if {@code start > end} or
150      *         if {@code start < 0 || end >= this.length() || otherStart < 0 ||
151      *         otherStart + (end - start) >= other.length()}
152      */
153     public default void swap(
154         final int start, final int end,
155         final MSeq<T> other, final int otherStart
156     ) {
157         if (otherStart < 0 || (otherStart + (end - start)) > length()) {
158             throw new ArrayIndexOutOfBoundsException(format(
159                 "Invalid index range: [%d, %d)",
160                 otherStart, otherStart + (end - start)
161             ));
162         }
163 
164         if (start < end) {
165             for (int i = end - start; --i >= 0;) {
166                 final T temp = get(start + i);
167                 set(start + i, other.get(otherStart + i));
168                 other.set(otherStart + i, temp);
169             }
170         }
171     }

173     /**
174      * Randomize the {@code array} using the {@link Random} object currently
175      * registered in the {@link RandomRegistry} class. The used shuffling
176      * algorithm is from D. Knuth TAOCP, Seminumerical Algorithms, Third edition,
177      * page 142, Algorithm S (Selection sampling technique).
178      *
179      * @return this shuffled sequence
180      */
181     public default MSeq<T> shuffle() {
182         return shuffle(RandomRegistry.getRandom());
183     }

185     /**
186      * Randomize the {@code array} using the given {@link Random} object. The used
187      * shuffling algorithm is from D. Knuth TAOCP, Seminumerical Algorithms,
188      * Third edition, page 142, Algorithm S (Selection sampling technique).
189      *
190      * @param random the {@link Random} object to use for randomize.
191      * @return this shuffled sequence
192      * @throws NullPointerException if the random object is {@code null}.
193      */
194     public default MSeq<T> shuffle(final Random random) {
195         for (int j = length() - 1; j > 0; --j) {
196             swap(j, random.nextInt(j + 1));
197         }
198         return this;
199     }

249     /**
250      * Returns a {@code Collector} that accumulates the input elements into a
251      * new {@code MSeq}.
252      *
253      * @param <T> the type of the input elements
254      * @return a {@code Collector} which collects all the input elements into a
255      *         {@code MSeq}, in encounter order
256      */
257     public static <T> Collector<T, ?, MSeq<T>> toMSeq() {
258         return Collector.of(
259             (Supplier<List<T>>)ArrayList::new,
260             List::add,
261             (left, right) -> { left.addAll(right); return left; },
262             MSeq::of
263         );
264     }

267      * Create a new {@code MSeq} with the given {@code length}.
268      *

281     /**
282      * Create a new {@code MSeq} from the given values.
283      *
284      * @param <T> the element type
285      * @param values the array values.
286      * @return a new {@code Meq} with the given values.
287      * @throws NullPointerException if the {@code values} array is {@code null}.
288      */
289     @SafeVarargs
290     public static <T> MSeq<T> of(final T... values) {
291         return values.length == 0
292             ? empty()
293             : new ArrayProxyMSeq<>(
294                 new ObjectArrayProxy<>(values.clone(), 0, values.length));
295     }

297     /**
298      * Create a new {@code MSeq} from the given values.
299      *
300      * @param <T> the element type
301      * @param values the array values.
302      * @return a new {@code MSeq} with the given values.
303      * @throws NullPointerException if the {@code values} object is
304      *        {@code null}.
305      */
306     @SuppressWarnings("unchecked")
307     public static <T> MSeq<T> of(final Iterable<? extends T> values) {
308         final MSeq<T> mseq;
309         if (values instanceof ISeq<?>) {
310             final ISeq<T> seq = (ISeq<T>)values;
311             mseq = seq.isEmpty() ? empty() : seq.copy();
312         } else if (values instanceof MSeq<?>) {
313             final MSeq<T> seq = (MSeq<T>)values;
314             mseq = seq.isEmpty() ? empty() : MSeq.of(seq);
315         } else if (values instanceof Collection<?>) {
316             final Collection<T> collection = (Collection<T>)values;
317             mseq = collection.isEmpty()
318                 ? empty()
319                 : MSeq.<T>ofLength(collection.size()).setAll(values);
320         } else {
321             final Stream.Builder<T> builder = Stream.builder();
322             values.forEach(builder::add);
323             final Object[] objects = builder.build().toArray();
324 
325             mseq = objects.length == 0
326                 ? empty()
327                 : new ArrayProxyMSeq<>(
328                     new ObjectArrayProxy<>(objects, 0, objects.length));
329         }
330 
331         return mseq;
332     }

356     /**
357      * Creates a new sequence, which is filled with objects created be the given
358      * {@code supplier}.
359      *
360      * @since 3.3
361      *
362      * @param <T> the element type of the sequence
363      * @param supplier the {@code Supplier} which creates the elements, the
364      *        returned sequence is filled with
365      * @param length the length of the returned sequence
366      * @return a new sequence filled with elements given by the {@code supplier}
367      * @throws NegativeArraySizeException if the given {@code length} is
368      *         negative
369      * @throws NullPointerException if the given {@code supplier} is
370      *         {@code null}
371      */
372     static <T> MSeq<T> of(
373         final Supplier<? extends T> supplier,
374         final int length
375     ) {
376         requireNonNull(supplier);
377 
378         return length == 0
379             ? empty()
380             : MSeq.<T>ofLength(length).fill(supplier);
381     }
