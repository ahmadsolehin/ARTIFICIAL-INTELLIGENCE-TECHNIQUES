20 package org.jenetics.internal.util;

22 import java.util.Random;
23 
24 import org.jenetics.util.RandomRegistry;
25 
26 /**
27  * Helper class which contains array helper methods.
28  *
29  * @author <a href="mailto:franz.wilhelmstoetter@gmx.at">Franz Wilhelmstötter</a>
30  * @since 3.0
31  * @version 3.0
32  */
33 public final class array {
34     private array() {require.noInstance();}
35 
36     public static void swap(final int[] array, final int i, final int j) {
37         final int temp = array[i];
38         array[i] = array[j];
39         array[j] = temp;
40     }
41 
42     public static void swap(final double[] array, final int i, final int j) {
43         final double temp = array[i];
44         array[i] = array[j];
45         array[j] = temp;
46     }
47 
48     public static int[] revert(final int[] array) {
49         for (int i = 0, j = array.length - 1; i < j; ++i, --j) {
50             swap(array, i, j);
51         }
52 
53         return array;
54     }
55 
56     public static double[] revert(final double[] array) {
57         for (int i = 0, j = array.length - 1; i < j; ++i, --j) {
58             swap(array, i, j);
59         }
60 
61         return array;
62     }
63 
64     /**
65      * Randomize the {@code array} using the given {@link Random} object. The used
66      * shuffling algorithm is from D. Knuth TAOCP, Seminumerical Algorithms,
67      * Third edition, page 142, Algorithm S (Selection sampling technique).
68      *
69      * @param array the array to shuffle
70      * @param random the PRNG
71      */
72     public static void shuffle(final double[] array, final Random random) {
73         for (int j = array.length - 1; j > 0; --j) {
74             swap(array, j, random.nextInt(j + 1));
75         }
76     }
77 
78     public static void shuffle(final double[] array) {
79         shuffle(array, RandomRegistry.getRandom());
80     }
