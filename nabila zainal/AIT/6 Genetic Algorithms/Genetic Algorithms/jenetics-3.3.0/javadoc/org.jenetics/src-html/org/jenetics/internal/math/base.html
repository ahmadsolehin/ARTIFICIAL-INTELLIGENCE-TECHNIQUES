042     /**
043      * <i>Clamping</i> a value between a pair of boundary values.
044      * <i>Note: using clamp with floating point numbers may give unexpected
045      * results if one of the values is {@code NaN}.</i>
046      *
047      * @param v the value to <i>clamp</i>
048      * @param lo the lower bound.
049      * @param hi the upper bound.
050      * @return The clamped value:
051      *        <ul>
052      *            <li>{@code lo if v < lo}</li>
053      *            <li>{@code hi if hi < v}</li>
054      *            <li>{@code otherwise, v}</li>
055      *        </ul>
056      */
057     public static double clamp(final double v, final double lo, final double hi) {
058         return v < lo ? lo : (v > hi ? hi : v);
059     }

061     /**
062      * Return the <a href="http://en.wikipedia.org/wiki/Unit_in_the_last_place">ULP</a>
063      * distance of the given two double values.
064      *
065      * @param a first double.
066      * @param b second double.
067      * @return the ULP distance.
068      * @throws ArithmeticException if the distance doesn't fit in a long value.
069      */
070     public static long ulpDistance(final double a, final double b) {
071         return Math.subtractExact(ulpPosition(a), ulpPosition(b));
072     }

074     /**
075      * Calculating the <a href="http://en.wikipedia.org/wiki/Unit_in_the_last_place">ULP</a>
076      * position of a double number.
077      *
078      * <pre>{@code
079      * double a = 0.0;
080      * for (int i = 0; i < 10; ++i) {
081      *     a = Math.nextAfter(a, Double.POSITIVE_INFINITY);
082      * }
083      *
084      * for (int i = 0; i < 19; ++i) {
085      *     a = Math.nextAfter(a, Double.NEGATIVE_INFINITY);
086      *     System.out.println(
087      *          a + "\t" + ulpPosition(a) + "\t" + ulpDistance(0.0, a)
088      *     );
089      * }
090      * }</pre>
091      *
092      * The code fragment above will create the following output:
093      * <pre>
094      *   4.4E-323    9  9
095      *   4.0E-323    8  8
096      *   3.5E-323    7  7
097      *   3.0E-323    6  6
098      *   2.5E-323    5  5
099      *   2.0E-323    4  4
100      *   1.5E-323    3  3
101      *   1.0E-323    2  2
102      *   4.9E-324    1  1
103      *   0.0         0  0
104      *  -4.9E-324   -1  1
105      *  -1.0E-323   -2  2
106      *  -1.5E-323   -3  3
107      *  -2.0E-323   -4  4
108      *  -2.5E-323   -5  5
109      *  -3.0E-323   -6  6
110      *  -3.5E-323   -7  7
111      *  -4.0E-323   -8  8
112      *  -4.4E-323   -9  9
113      * </pre>
114      *
115      * @param a the double number.
116      * @return the ULP position.
117      */
118     public static long ulpPosition(final double a) {
119         long t = doubleToLongBits(a);
120         if (t < 0) {
121             t = Long.MIN_VALUE - t;
122         }
123         return t;
124     }

142      * Selects a random subset of size {@code k} from a set of size {@code n}.
143      *
144      * @see #subset(int, int[], Random)
145      *
146      * @param n the size of the set.
147      * @param k the size of the subset.
148      * @param random the random number generator used.
149      * @throws NullPointerException if {@code random} is {@code null}.
150      * @throws IllegalArgumentException if {@code n < k}, {@code k == 0} or if
151      *         {@code n*k} will cause an integer overflow.
152      * @return the subset array.
153      */
154     public static int[] subset(final int n, final int k, final Random random) {
155         requireNonNull(random, "Random");
156         if (k <= 0) {
157             throw new IllegalArgumentException(format(
158                     "Subset size smaller or equal zero: %s", k
159                 ));
160         }
161         if (n < k) {
162             throw new IllegalArgumentException(format(
163                     "n smaller than k: %s < %s.", n, k
164                 ));
165         }
166 
167         final int[] sub = new int[k];
168         subset(n, sub,random);
169         return sub;
170     }

205     /**
206      * <p>
207      * Selects a random subset of size {@code sub.length} from a set of size
208      * {@code n}.
209      * </p>
210      *
211      * <p>
212      * <em>Authors:</em>
213      *      FORTRAN77 original version by Albert Nijenhuis, Herbert Wilf. This
214      *      version based on the  C++ version by John Burkardt.
215      * </p>
216      *
217      * <p><em><a href="https://people.scs.fsu.edu/~burkardt/c_src/subset/subset.html">
218      *  Reference:</a></em>
219      *      Albert Nijenhuis, Herbert Wilf,
220      *      Combinatorial Algorithms for Computers and Calculators,
221      *      Second Edition,
222      *      Academic Press, 1978,
223      *      ISBN: 0-12-519260-6,
224      *      LC: QA164.N54.
225      * </p>
226      *
227      * @param n the size of the set.
228      * @param sub the sub set array.
229      * @param random the random number generator used.
230      * @return the sub-set array for the given parameter
231      * @throws NullPointerException if {@code sub} or {@code random} is
232      *         {@code null}.
233      * @throws IllegalArgumentException if {@code n < sub.length},
234      *         {@code sub.length == 0} or {@code n*sub.length} will cause an
235      *         integer overflow.
236      */
237     public static int[] subset(final int n, final int sub[], final Random random) {
238         requireNonNull(random, "Random");
239         requireNonNull(sub, "Sub set array");
240 
241         final int k = sub.length;
242         if (k <= 0) {
243             throw new IllegalArgumentException(format(
244                 "Subset size smaller or equal zero: %s", k
245             ));
246         }
247         if (n < k) {
248             throw new IllegalArgumentException(format(
249                 "n smaller than k: %s < %s.", n, k
250             ));
251         }
252         if (!arithmetic.isMultiplicationSave(n, k)) {
253             throw new IllegalArgumentException(format(
254                 "n*sub.length > Integer.MAX_VALUE (%s*%s = %s > %s)",
255                 n, sub.length, (long)n*(long)k, Integer.MAX_VALUE
256             ));
257         }
258 
259         if (sub.length == n) {
260             for (int i = 0; i < sub.length; ++i) {
261                 sub[i] = i;
262             }
263             return sub;
264         }
265 
266         for (int i = 0; i < k; ++i) {
267             sub[i] = (i*n)/k;
268         }
269 
270         int l = 0;
271         int ix = 0;
272         for (int i = 0; i < k; ++i) {
273             do {
274                 ix = nextInt(random, 1, n);
275                 l = (ix*k - 1)/n;
276             } while (sub[l] >= ix);
277 
278             sub[l] = sub[l] + 1;
279         }
280 
281         int m = 0;
282         int ip = 0;
283         int is = k;
284         for (int i = 0; i < k; ++i) {
285             m = sub[i];
286             sub[i] = 0;
287 
288             if (m != (i*n)/k) {
289                 ip = ip + 1;
290                 sub[ip - 1] = m;
291             }
292         }
293 
294         int ihi = ip;
295         int ids = 0;
296         for (int i = 1; i <= ihi; ++i) {
297             ip = ihi + 1 - i;
298             l = 1 + (sub[ip - 1]*k - 1)/n;
299             ids = sub[ip - 1] - ((l - 1)*n)/k;
300             sub[ip - 1] = 0;
301             sub[is - 1] = l;
302             is = is - ids;
303         }
304 
305         int ir = 0;
306         int m0 = 0;
307         for (int ll = 1; ll <= k; ++ll) {
308             l = k + 1 - ll;
309 
310             if (sub[l - 1] != 0) {
311                 ir = l;
312                 m0 = 1 + ((sub[l - 1] - 1)*n)/k;
313                 m = (sub[l-1]*n)/k - m0 + 1;
314             }
315 
316             ix = nextInt(random, m0, m0 + m - 1);
317 
318             int i = l + 1;
319             while (i <= ir && ix >= sub[i - 1]) {
320                 ix = ix + 1;
321                 sub[ i- 2] = sub[i - 1];
322                 i = i + 1;
323             }
324 
325             sub[i - 2] = ix;
326             --m;
327         }
328 
329         return sub;
330     }

332     private static int nextInt(final Random random, final int a, final int b) {
333         return a == b ? a - 1 : random.nextInt(b - a) + a;
334     }
