057/**
058 * Genetic algorithm <em>engine</em> which is the main class. The following
059 * example shows the main steps in initializing and executing the GA.
060 *
061 * <pre>{@code
062 * public class RealFunction {
063 *    // Definition of the fitness function.
064 *    private static Double eval(final Genotype<DoubleGene> gt) {
065 *        final double x = gt.getGene().doubleValue();
066 *        return cos(0.5 + sin(x)) * cos(x);
067 *    }
068 *
069 *    public static void main(String[] args) {
070 *        // Create/configuring the engine via its builder.
071 *        final Engine<DoubleGene, Double> engine = Engine
072 *            .builder(
073 *                RealFunction::eval,
074 *                DoubleChromosome.of(0.0, 2.0*PI))
075 *            .populationSize(500)
076 *            .optimize(Optimize.MINIMUM)
077 *            .alterers(
078 *                new Mutator<>(0.03),
079 *                new MeanAlterer<>(0.6))
080 *            .build();
081 *
082 *        // Execute the GA (engine).
083 *        final Phenotype<DoubleGene, Double> result = engine.stream()
084 *             // Truncate the evolution stream if no better individual could
085 *             // be found after 5 consecutive generations.
086 *            .limit(bySteadyFitness(5))
087 *             // Terminate the evolution after maximal 100 generations.
088 *            .limit(100)
089 *            .collect(toBestPhenotype());
090 *     }
091 * }
092 * }</pre>
093 *
094 * The architecture allows to decouple the configuration of the engine from the
095 * execution. The {@code Engine} is configured via the {@code Engine.Builder}
096 * class and can't be changed after creation. The actual <i>evolution</i> is
097 * performed by the {@link EvolutionStream}, which is created by the
098 * {@code Engine}.
099 * <p>
100 * <em>
101 *     <b>This class is thread safe:</b>
102 *     No mutable state is maintained by the engine. Therefore it is save to
103 *     create multiple evolution streams with one engine, which may be actually
104 *     used in different threads.
105 * </em>
106 *
107 * @see Engine.Builder
108 * @see EvolutionStart
109 * @see EvolutionResult
110 * @see EvolutionStream
111 * @see EvolutionStatistics
112 * @see Codec
113 *
114 * @author <a href="mailto:franz.wilhelmstoetter@gmx.at">Franz Wilhelmstötter</a>
115 * @since 3.0
116 * @version 3.2
117 */
118public final class Engine<

120        C extends Comparable<? super C>
121>

146        /**
147         * Create a new GA engine with the given parameters.
148         *
149         * @param genotypeFactory the genotype factory this GA is working with.
150         * @param fitnessFunction the fitness function this GA is using.
151         * @param fitnessScaler the fitness scaler this GA is using.
152         * @param survivorsSelector the selector used for selecting the survivors
153         * @param offspringSelector the selector used for selecting the offspring
154         * @param alterer the alterer used for altering the offspring
155         * @param validator phenotype validator which can override the default
156         *        implementation the {@link Phenotype#isValid()} method.
157         * @param optimize the kind of optimization (minimize or maximize)
158         * @param offspringCount the number of the offspring individuals
159         * @param survivorsCount the number of the survivor individuals
160         * @param maximalPhenotypeAge the maximal age of an individual
161         * @param executor the executor used for executing the single evolve steps
162         * @param clock the clock used for calculating the timing results
163         * @param individualCreationRetries the maximal number of attempts for
164         *        creating a valid individual.
165         * @throws NullPointerException if one of the arguments is {@code null}
166         * @throws IllegalArgumentException if the given integer values are smaller
167         *         than one.
168         */
169        Engine(
170                final Function<? super Genotype<G>, ? extends C> fitnessFunction,
171                final Function<? super C, ? extends C> fitnessScaler,
172                final Factory<Genotype<G>> genotypeFactory,
173                final Selector<G, C> survivorsSelector,
174                final Selector<G, C> offspringSelector,
175                final Alterer<G, C> alterer,
176                final Predicate<? super Phenotype<G, C>> validator,
177                final Optimize optimize,
178                final int offspringCount,
179                final int survivorsCount,
180                final long maximalPhenotypeAge,
181                final Executor executor,
182                final Clock clock,
183                final int individualCreationRetries
184        ) {
185                _fitnessFunction = requireNonNull(fitnessFunction);
186                _fitnessScaler = requireNonNull(fitnessScaler);
187                _genotypeFactory = requireNonNull(genotypeFactory);
188                _survivorsSelector = requireNonNull(survivorsSelector);
189                _offspringSelector = requireNonNull(offspringSelector);
190                _alterer = requireNonNull(alterer);
191                _validator = requireNonNull(validator);
192                _optimize = requireNonNull(optimize);
193
194                _offspringCount = require.positive(offspringCount);
195                _survivorsCount = require.positive(survivorsCount);
196                _maximalPhenotypeAge = require.positive(maximalPhenotypeAge);
197
198                _executor = new TimedExecutor(requireNonNull(executor));
199                _clock = requireNonNull(clock);
200
201                if (individualCreationRetries < 0) {
202                        throw new IllegalArgumentException(format(
203                                "Retry count must not be negative: %d",
204                                individualCreationRetries
205                        ));
206                }
207                _individualCreationRetries = individualCreationRetries;
208        }

211         * Perform one evolution step with the given {@code population} and
212         * {@code generation}. New phenotypes are created with the fitness function
213         * and fitness scaler defined by this <em>engine</em>
214         * <p>
215         * <em>This method is thread-safe.</em>
216         *
217         * @see #evolve(EvolutionStart)
218         *
219         * @param population the population to evolve
220         * @param generation the current generation; used for calculating the
221         *        phenotype age.
222         * @return the evolution result
223         * @throws java.lang.NullPointerException if the given {@code population} is
224         *         {@code null}
225         * @throws IllegalArgumentException if the given {@code generation} is
226         *         smaller then one
227         */
228        public EvolutionResult<G, C> evolve(
229                final Population<G, C> population,
230                final long generation
231        ) {
232                return evolve(EvolutionStart.of(population, generation));
233        }

235        /**
236         * Perform one evolution step with the given evolution {@code start} object
237         * New phenotypes are created with the fitness function and fitness scaler
238         * defined by this <em>engine</em>
239         * <p>
240         * <em>This method is thread-safe.</em>
241         *
242         * @since 3.1
243         * @see #evolve(org.jenetics.Population, long)
244         *
245         * @param start the evolution start object
246         * @return the evolution result
247         * @throws java.lang.NullPointerException if the given evolution
248         *         {@code start} is {@code null}
249         */
250        public EvolutionResult<G, C> evolve(final EvolutionStart<G, C> start) {
251                final Timer timer = Timer.of().start();
252
253                // Initial evaluation of the population.
254                final Timer evaluateTimer = Timer.of(_clock).start();
255                evaluate(start.getPopulation());
256                evaluateTimer.stop();
257
258                // Select the offspring population.
259                final CompletableFuture<TimedResult<Population<G, C>>> offspring =
260                        _executor.async(() ->
261                                selectOffspring(start.getPopulation()),
262                                _clock
263                        );
264
265                // Select the survivor population.
266                final CompletableFuture<TimedResult<Population<G, C>>> survivors =
267                        _executor.async(() ->
268                                selectSurvivors(start.getPopulation()),
269                                _clock
270                        );
271
272                // Altering the offspring population.
273                final CompletableFuture<TimedResult<AlterResult<G, C>>> alteredOffspring =
274                        _executor.thenApply(offspring, p ->
275                                alter(p.result, start.getGeneration()),
276                                _clock
277                        );
278
279                // Filter and replace invalid and to old survivor individuals.
280                final CompletableFuture<TimedResult<FilterResult<G, C>>> filteredSurvivors =
281                        _executor.thenApply(survivors, pop ->
282                                filter(pop.result, start.getGeneration()),
283                                _clock
284                        );
285
286                // Filter and replace invalid and to old offspring individuals.
287                final CompletableFuture<TimedResult<FilterResult<G, C>>> filteredOffspring =
288                        _executor.thenApply(alteredOffspring, pop ->
289                                filter(pop.result.population, start.getGeneration()),
290                                _clock
291                        );
292
293                // Combining survivors and offspring to the new population.
294                final CompletableFuture<Population<G, C>> population =
295                        filteredSurvivors.thenCombineAsync(filteredOffspring, (s, o) -> {
296                                        final Population<G, C> pop = s.result.population;
297                                        pop.addAll(o.result.population);
298                                        return pop;
299                                },
300                                _executor.get()
301                        );
302
303                // Evaluate the fitness-function and wait for result.
304                final TimedResult<Population<G, C>> result = population
305                        .thenApply(TimedResult.of(this::evaluate, _clock))
306                        .join();
307
308                final EvolutionDurations durations = EvolutionDurations.of(
309                        offspring.join().duration,
310                        survivors.join().duration,
311                        alteredOffspring.join().duration,
312                        filteredOffspring.join().duration,
313                        filteredSurvivors.join().duration,
314                        result.duration.plus(evaluateTimer.getTime()),
315                        timer.stop().getTime()
316                );
317
318                final int killCount =
319                        filteredOffspring.join().result.killCount +
320                        filteredSurvivors.join().result.killCount;
321
322                final int invalidCount =
323                        filteredOffspring.join().result.invalidCount +
324                        filteredSurvivors.join().result.invalidCount;
325
326                return EvolutionResult.of(
327                        _optimize,
328                        result.result,
329                        start.getGeneration(),
330                        durations,
331                        killCount,
332                        invalidCount,
333                        alteredOffspring.join().result.alterCount
334                );
335        }

358        // Filters out invalid and to old individuals. Filtering is done in place.
359        private FilterResult<G, C> filter(
360                final Population<G, C> population,
361                final long generation
362        ) {
363                int killCount = 0;
364                int invalidCount = 0;
365
366                for (int i = 0, n = population.size(); i < n; ++i) {
367                        final Phenotype<G, C> individual = population.get(i);
368
369                        if (!_validator.test(individual)) {
370                                population.set(i, newPhenotype(generation));
371                                ++invalidCount;
372                        } else if (individual.getAge(generation) > _maximalPhenotypeAge) {
373                                population.set(i, newPhenotype(generation));
374                                ++killCount;
375                        }
376                }
377
378                return new FilterResult<>(population, killCount, invalidCount);
379        }
380
381        // Create a new and valid phenotype
382        private Phenotype<G, C> newPhenotype(final long generation) {
383                int count = 0;
384                Phenotype<G, C> phenotype;
385                do {
386                        phenotype = Phenotype.of(
387                                _genotypeFactory.newInstance(),
388                                generation,
389                                _fitnessFunction,
390                                _fitnessScaler
391                        );
392                } while (++count < _individualCreationRetries &&
393                                !_validator.test(phenotype));
394
395                return phenotype;
396        }

398        // Alters the given population. The altering is done in place.
399        private AlterResult<G, C> alter(
400                final Population<G,C> population,
401                final long generation
402        ) {
403                return new AlterResult<>(
404                        population,
405                        _alterer.alter(population, generation)
406                );

409        // Evaluates the fitness function of the give population concurrently.
410        private Population<G, C> evaluate(final Population<G, C> population) {
411                try (Concurrency c = Concurrency.with(_executor.get())) {
412                        c.execute(population);
413                }
414                return population;
415        }

441        private EvolutionStart<G, C> evolutionStart() {
442                final int generation = 1;
443                final int size = _offspringCount + _survivorsCount;
444
445                final Population<G, C> population = new Population<G, C>(size)
446                        .fill(() -> newPhenotype(generation), size);
447
448                return EvolutionStart.of(population, generation);
449        }

497        private EvolutionStart<G, C> evolutionStart(
498                final Iterable<Genotype<G>> genotypes,
499                final long generation
500        ) {
501                final Stream<Phenotype<G, C>> stream = Stream.concat(
502                        StreamSupport.stream(genotypes.spliterator(), false)
503                                .map(gt -> Phenotype.of(
504                                        gt, generation, _fitnessFunction, _fitnessScaler)),
505                        Stream.generate(() -> newPhenotype(generation))
506                );
507
508                final Population<G, C> population = stream
509                        .limit(getPopulationSize())
510                        .collect(toPopulation());
511
512                return EvolutionStart.of(population, generation);
513        }

579        private EvolutionStart<G, C> evolutionStart(
580                final Population<G, C> population,
581                final long generation
582        ) {
583                final Stream<Phenotype<G, C>> stream = Stream.concat(
584                        population.stream()
585                                .map(p -> p.newInstance(
586                                        p.getGeneration(),
587                                        _fitnessFunction,
588                                        _fitnessScaler)),
589                        Stream.generate(() -> newPhenotype(generation))
590                );
591
592                final Population<G, C> pop = stream
593                        .limit(getPopulationSize())
594                        .collect(toPopulation());
595
596                return EvolutionStart.of(pop, generation);
597        }

734         * engine can serve as a template for an new one.
735         *
736         * @return a new engine builder
737         */
738        public Builder<G, C> builder() {
739                return new Builder<>(_genotypeFactory, _fitnessFunction)
740                        .alterers(_alterer)
741                        .clock(_clock)
742                        .executor(_executor.get())
743                        .fitnessScaler(_fitnessScaler)
744                        .maximalPhenotypeAge(_maximalPhenotypeAge)
745                        .offspringFraction((double)_offspringCount/(double)getPopulationSize())
746                        .offspringSelector(_offspringSelector)
747                        .optimize(_optimize)
748                        .phenotypeValidator(_validator)
749                        .populationSize(getPopulationSize())
750                        .survivorsSelector(_survivorsSelector)
751                        .individualCreationRetries(_individualCreationRetries);
752        }
