055 /**
056  * A population is a collection of Phenotypes.
057  *
058  * <p>
059  * <strong>This class is not synchronized.</strong> If multiple threads access
060  * a {@code Population} concurrently, and at least one of the threads modifies
061  * it, it <strong>must</strong> be synchronized externally.
062  *
063  * @author <a href="mailto:franz.wilhelmstoetter@gmx.at">Franz Wilhelmstötter</a>
064  * @since 1.0
065  * @version 2.0
066  */

118     /**
119      * Fills the population with individuals created by the given factory.
120      *
121      * @param factory the {@code Phenotype} factory.
122      * @param count the number of individuals to add to this population.
123      * @return return this population, for command chaining.
124      */
125     public Population<G, C> fill(
126         final Factory<Phenotype<G, C>> factory,
127         final int count
128     ) {
129         for (int i = 0; i < count; ++i) {
130             _population.add(factory.newInstance());
131         }
132         return this;
133     }

329     /**
330      * Returns a {@code Collector} that accumulates the input elements into a
331      * new {@code Population}.
332      *
333      * @param <G> the gene type
334      * @param <C> the fitness result type
335      * @return a {@code Collector} which collects all the input elements into a
336      *         {@code Population}, in encounter order
337      */
338     public static <G extends Gene<?, G>, C extends Comparable<? super C>>
339     Collector<Phenotype<G, C>, ?, Population<G, C>> toPopulation() {
340         return Collector.of(
341             Population::new,
342             Population::add,
343             (left, right) -> { left.addAll(right); return left; }
344         );
345     }

351     @XmlRootElement(name = "population")
352     @XmlType(name = "org.jenetics.Population")
353     @XmlAccessorType(XmlAccessType.FIELD)
354     @SuppressWarnings({"unchecked", "rawtypes"})
355     static final class Model {
356 
357         @XmlAttribute(name = "size", required = true)
358         public int size;
359 
360         @XmlElement(name = "phenotype", required = true)
361         public List phenotypes;
362 
363         public static final class Adapter
364             extends XmlAdapter<Model, Population>
365         {
366             @Override
367             public Model marshal(final Population p) throws Exception {
368                 final Model model = new Model();
369                 model.size = p.size();
370                 if (!p.isEmpty()) {
371                     model.phenotypes = (List)p.stream()
372                         .map(jaxb.Marshaller(p.get(0)))
373                         .collect(toList());
374                 }
375 
376                 return model;
377             }
378 
379             @Override
380             public Population unmarshal(final Model model) throws Exception {
381                 return (Population)model.phenotypes.stream()
382                     .map(jaxb.Unmarshaller(model.phenotypes.get(0)))
383                     .collect(toPopulation());
384             }
385         }
