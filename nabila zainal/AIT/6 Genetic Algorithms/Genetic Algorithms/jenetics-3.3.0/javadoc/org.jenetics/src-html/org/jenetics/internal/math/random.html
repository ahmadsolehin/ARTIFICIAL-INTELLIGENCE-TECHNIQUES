061     /**
062      * Returns a pseudo-random, uniformly distributed int value between min and
063      * max (min and max included).
064      *
065      * @param random the random engine to use for calculating the random int
066      *        value
067      * @param min lower bound for generated integer
068      * @param max upper bound for generated integer
069      * @return a random integer greater than or equal to {@code min} and
070      *         less than or equal to {@code max}
071      * @throws IllegalArgumentException if {@code min > max}
072      * @throws NullPointerException if the given {@code random}
073      *         engine is {@code null}.
074      */
075     public static int nextInt(
076         final Random random,
077         final int min, final int max
078     ) {
079         if (min > max) {
080             throw new IllegalArgumentException(format(
081                 "Min >= max: %d >= %d", min, max
082             ));
083         }
084 
085         final int diff = max - min + 1;
086         int result = 0;
087 
088         if (diff <= 0) {
089             do {
090                 result = random.nextInt();
091             } while (result < min || result > max);
092         } else {
093             result = random.nextInt(diff) + min;
094         }
095 
096         return result;
097     }

101      * and max (min and max included).
102      *
103      * @param random the random engine to use for calculating the random
104      *        long value
105      * @param min lower bound for generated long integer
106      * @param max upper bound for generated long integer
107      * @return a random long integer greater than or equal to {@code min}
108      *         and less than or equal to {@code max}
109      * @throws IllegalArgumentException if {@code min > max}
110      * @throws NullPointerException if the given {@code random}
111      *         engine is {@code null}.
112      */
113     public static long nextLong(
114         final Random random,
115         final long min, final long max
116     ) {
117         if (min > max) {
118             throw new IllegalArgumentException(format(
119                 "min >= max: %d >= %d.", min, max
120             ));
121         }
122

124         long result = 0;

129             } while (result < min || result > max);
130         } else if (diff < Integer.MAX_VALUE) {
131             result = random.nextInt((int)diff) + min;
132         } else {

134         }

137     }

140      * Returns a pseudo-random, uniformly distributed int value between 0
141      * (inclusive) and the specified value (exclusive), drawn from the given
142      * random number generator's sequence.
143      *
144      * @param random the random engine used for creating the random number.
145      * @param n the bound on the random number to be returned. Must be
146      *        positive.
147      * @return the next pseudo-random, uniformly distributed int value
148      *         between 0 (inclusive) and n (exclusive) from the given random
149      *         number generator's sequence
150      * @throws IllegalArgumentException if n is smaller than 1.
151      * @throws NullPointerException if the given {@code random}
152      *         engine is {@code null}.
153      */
154     public static long nextLong(final Random random, final long n) {
155         if (n <= 0) {
156             throw new IllegalArgumentException(format(
157                 "n is smaller than one: %d", n
158             ));
159         }
160 
161         long bits;
162         long result;
163         do {
164             bits = random.nextLong() & 0x7fffffffffffffffL;
165             result = bits%n;
166         } while (bits - result + (n - 1) < 0);
167 
168         return result;
169     }

241      *
242      * @param random the random engine used for creating the random number.
243      * @param n the bound on the random number to be returned. Must be
244      *        positive.
245      * @return the next pseudo-random, uniformly distributed int value
246      *         between 0 (inclusive) and n (exclusive) from the given random
247      *         number generator's sequence
248      * @throws IllegalArgumentException if n is smaller than 1.
249      * @throws NullPointerException if the given {@code random}
250      *         engine of the maximal value {@code n} is {@code null}.
251      */
252     public static BigInteger nextBigInteger(
253         final Random random,
254         final BigInteger n
255     ) {
256         if (n.compareTo(BigInteger.ONE) < 0) {
257             throw new IllegalArgumentException(format(
258                 "n is smaller than one: %d", n
259             ));
260         }
261 
262         BigInteger result = null;
263         if (n.bitLength() <= Integer.SIZE - 1) {
264             result = BigInteger.valueOf(random.nextInt(n.intValue()));
265         } else if (n.bitLength() <= Long.SIZE - 1) {
266             result = BigInteger.valueOf(nextLong(random, n.longValue()));
267         } else {
268             do {
269                 result = new BigInteger(n.bitLength(), random);
270             } while (result.compareTo(n) >= 0);
271         }
272 
273         return result;
274     }

376      * given range and the index probability.
377      *
378      * @since 3.0
379      *
380      * @param random the random engine used for calculating the random
381      *        indexes
382      * @param start the start index (inclusively)

385      * @return an new random index stream
386      * @throws IllegalArgumentException if {@code p} is not a
387      *         valid probability.
388      */
389     public static IntStream indexes(
390         final Random random,
391         final int start,
392         final int end,
393         final double p
394     ) {
395         probability(p);
396         final int P = probability.toInt(p);
397 
398         return equals(p, 0, 1E-20)
399             ? IntStream.empty()
400             : equals(p, 1, 1E-20)
401                 ? IntStream.range(start, end)
402                 : IntStream.range(start, end)
403                     .filter(i -> random.nextInt() < P);
404     }

452      * calls of the {@link #seed()} method.
453      *
454      * @see #seed()
455      *
456      * @param seed the byte array seed to fill with random bytes.
457      * @return the given byte array, for method chaining.

462         for (int i = 0, len = seed.length; i < len;) {
463             int n = Math.min(len - i, Long.SIZE/Byte.SIZE);
464 
465             for (long x = seed(); n-- > 0; x >>= Byte.SIZE) {
466                 seed[i++] = (byte)x;
467             }
468         }
469

471     }

473     /**
474      * Calculating a 64 bit seed value which can be used for initializing
475      * PRNGs. This method uses a combination of {@code System.nanoTime()}
476      * and {@code new Object().hashCode()} calls to create a reasonable safe
477      * seed value:
478      * <p>
479      * <pre>{@code
480      * public static long seed() {
481      *     return seed(System.nanoTime());
482      * }
483      * }</pre>
484      * <p>
485      * This method passes all of the statistical tests of the
486      * <a href="http://www.phy.duke.edu/~rgb/General/dieharder.php">
487      * dieharder</a> test suite&mdash;executed on a linux machine with
488      * JDK version 1.7. <em>Since there is no prove that this will the case
489      * for every Java version and OS, it is recommended to only use this
490      * method for seeding other PRNGs.</em>
491      *
492      * @see #seed(long)
493      *
494      * @return the random seed value.
495      */
496     public static long seed() {
497         return seed(System.nanoTime());
498     }

500     /**
501      * Uses the given {@code base} value to create a reasonable safe seed
502      * value. This is done by combining it with values of
503      * {@code new Object().hashCode()}:
504      * <p>
505      * <pre>{@code
506      * public static long seed(final long base) {
