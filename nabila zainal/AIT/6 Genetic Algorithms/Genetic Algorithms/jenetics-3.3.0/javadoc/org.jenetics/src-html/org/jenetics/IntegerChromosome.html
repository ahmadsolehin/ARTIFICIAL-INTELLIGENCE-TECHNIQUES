044 /**
045  * Numeric chromosome implementation which holds 32 bit integer numbers.
046  *
047  * @author <a href="mailto:franz.wilhelmstoetter@gmx.at">Franz  Wilhelmstötter</a>

049  * @version 3.2
050  */
051 @XmlJavaTypeAdapter(IntegerChromosome.Model.Adapter.class)
052 public class IntegerChromosome
053     extends AbstractBoundedChromosome<Integer, IntegerGene>
054     implements
055             NumericChromosome<Integer, IntegerGene>,
056             Serializable

061     protected IntegerChromosome(final ISeq<IntegerGene> genes) {
062         super(genes);

065     /**
066      * Create a new random {@code IntegerChromosome}.
067      *
068      * @param min the min value of the {@link IntegerGene}s (inclusively).
069      * @param max the max value of the {@link IntegerGene}s (inclusively).
070      * @param length the length of the chromosome.
071      * @throws NullPointerException if one of the arguments is {@code null}.
072      */
073     public IntegerChromosome(final Integer min, final Integer max, final int length) {
074         this(IntegerGene.seq(min, max, length));
075         _valid = true;
076     }
077 
078     /**
079      * Create a new random {@code IntegerChromosome} of length one.
080      *
081      * @param min the minimal value of this chromosome (inclusively).
082      * @param max the maximal value of this chromosome (inclusively).
083      * @throws NullPointerException if one of the arguments is {@code null}.
084      */
085     public IntegerChromosome(final Integer min, final Integer max) {
086         this(min, max, 1);
087     }
088 
089     /**
090      * Returns an int array containing all of the elements in this chromosome
091      * in proper sequence.  If the chromosome fits in the specified array, it is
092      * returned therein. Otherwise, a new array is allocated with the length of
093      * this chromosome.
094      *
095      * @since 3.0
096      *
097      * @param array the array into which the elements of this chromosomes are to
098      *        be stored, if it is big enough; otherwise, a new array is
099      *        allocated for this purpose.
100      * @return an array containing the elements of this chromosome
101      * @throws NullPointerException if the given {@code array} is {@code null}
102      */
103     public int[] toArray(final int[] array) {
104         final int[] a = array.length >= length() ?
105             array : new int[length()];
106 
107         for (int i = length(); --i >= 0;) {
108             a[i] = intValue(i);
109         }
110 
111         return a;
112     }
113 
114     /**
115      * Returns an int array containing all of the elements in this chromosome
116      * in proper sequence.
117      *
118      * @since 3.0
119      *
120      * @return an array containing the elements of this chromosome
121      */
122     public int[] toArray() {
123         return toArray(new int[length()]);
124     }

216     private void writeObject(final ObjectOutputStream out)
217         throws IOException
218     {
219         out.defaultWriteObject();
220 
221         out.writeInt(length());
222         out.writeInt(_min.intValue());
223         out.writeInt(_max.intValue());
224 
225         for (IntegerGene gene : _genes) {
226             out.writeInt(gene.getAllele().intValue());
227         }
228     }
229 
230     private void readObject(final ObjectInputStream in)
231         throws IOException, ClassNotFoundException
232     {
233         in.defaultReadObject();
234 
235         final MSeq<IntegerGene> genes = MSeq.ofLength(in.readInt());
236         _min = in.readInt();
237         _max = in.readInt();
238 
239         for (int i = 0; i < genes.length(); ++i) {
240             genes.set(i, new IntegerGene(in.readInt(), _min, _max));
241         }
242 
243         _genes = genes.toISeq();
244     }

250     @XmlRootElement(name = "int-chromosome")

253     final static class Model {

255         @XmlAttribute(name = "length", required = true)
256         public int length;
257 
258         @XmlAttribute(name = "min", required = true)

260 
261         @XmlAttribute(name = "max", required = true)
262         public int max;
263 
264         @XmlElement(name = "allele", required = true, nillable = false)
265         public List<Integer> values;

267         public final static class Adapter
268             extends XmlAdapter<Model, IntegerChromosome>
269         {
270             @Override
271             public Model marshal(final IntegerChromosome c) {
272                 final Model m = new Model();
273                 m.length = c.length();
274                 m.min = c._min;
275                 m.max = c._max;
276                 m.values = c.toSeq().map(IntegerGene::getAllele).asList();
277                 return m;
278             }
279 
280             @Override
281             public IntegerChromosome unmarshal(final Model model) {
282                 final Integer min = model.min;
283                 final Integer max = model.max;
284                 return new IntegerChromosome(
285                     ISeq.of(model.values)
286                         .map(a -> new IntegerGene(a, min, max))
287                 );
288             }
289         }
